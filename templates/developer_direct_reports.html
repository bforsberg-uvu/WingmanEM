Listing uses _db_load_direct_reports(): select(DirectReportORM).order_by(DirectReportORM.id), each row mapped to the same dict shape the UI expects. Persisting the in-memory list after adds/edits/deletes goes through _db_replace_direct_reports_from_list, which clears direct_reports and re-inserts one DirectReportORM per dict (primary key id preserved).

    def _next_direct_report_id() -> int:
    max_id = 0
    for r in direct_reports:
        max_id = max(max_id, int(r.get("id") or 0))
    return max_id + 1

# _db_replace_direct_reports_from_list (excerpt):
def _db_replace_direct_reports_from_list(reports: list[dict[str, Any]]) -> None:
    session = get_session()
    try:
        session.execute(delete(DirectReportORM))
        for r in reports:
            session.add(
                DirectReportORM(
                    id=int(r.get("id") or 0),
                    first_name=r.get("first_name") or "",
                    last_name=r.get("last_name") or "",
                    street_address_1=r.get("street_address_1"),
                    # … other mapped columns …
                )
            )
        session.commit()
    finally:
        session.close()
